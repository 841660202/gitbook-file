    }
}

文本消息添加链接：

{
    "touser": "oNdY2wReKoYCd8wx_vyicXazFl3A", 
    "msgtype": "text", 
    "text": {
        "content":"文本内容<a href=\"http://www.qq.com\"  >点击跳转qq网页链接</a>！"
    }
}
